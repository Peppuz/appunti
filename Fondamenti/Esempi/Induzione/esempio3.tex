\begin{thm}
 $\displaystyle \sum_{i = 1}^n 2i-1 = n^2$
\end{thm}

\begin{proof}
Per $n = 1 \quad \displaystyle \sum_{i = 1}^1 2i-1 = 1^2 \quad 1 = 1$ Ã¨ vero

Se $\displaystyle \sum_{i = 1}^n 2i-1 = n^2$ allora
$\displaystyle \sum_{i = 1}^{n+1} 2i-1 = (n+1)^2$

\begin{equation*}
\begin{split}
  \sum_{i=1}^{n+1} 2i-1 & = \sum_{i=1}^n 2i-1 + 2(n+1) - 1 \\
         & = n^2 + 2n + 1 \\
         & = (n+1)^2 \\
\end{split}
\end{equation*}
\end{proof}

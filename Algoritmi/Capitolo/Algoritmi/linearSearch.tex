\begin{algorithm}
\caption{\proc{linearSearch}(ITEM[] A,\textbf{integer} \id{key})}
\begin{codebox}
\li  \For $j = 0$ \To A.length-1
     \Do
\li           \If A[j] = key
              \Do
\li                    \Return A[j]
              \End
      \End
\li \Return \null
\end{codebox}
\end{algorithm}
